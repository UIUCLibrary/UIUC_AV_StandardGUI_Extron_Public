86 def __UpdatePINHandler(self, button: 'Button', action: str): 

87 if action == 'Pressed': 

88 button.SetState(1) 

89 elif action == 'Released': 

90 val = button.ID - 9000 

91 # pin button IDs should start at 9000 and be in numerical order 

92 self.__CurrentPIN = self.__CurrentPIN + str(val) 

93 self.__MaskPIN() #remask pin after change 

94 if (self.__CurrentPIN == self.PIN): 

95 self.UIHost.ShowPopup("PIN Outcome Success", 2) 

96 # clean up and go to destination page while success popup is up 

97 self.UIHost.ShowPage(self.__DestPage) 

98 self.__DestPageFn() 

99 self.UIHost.HidePopup("PIN Code") 

100 elif (len(self.__CurrentPIN) >= 10): 

101 self.UIHost.ShowPopup("PIN Outcome Failure", 2) 

102 # clean up and go back to pin page while failure popup is up 

103 self.ResetPIN() 

104 button.SetState(0) 

105 

106 def __BackspacePINHandler(self, button: 'Button', action: str): 

107 if action == 'Pressed': 

108 button.SetState(1) 

109 elif action == 'Released': 

110 self.__CurrentPIN = self.__CurrentPIN[:-1] # remove last character of current pin 

111 self.__MaskPIN() # remask pin after change 

112 button.SetState(0) 

113 

114 def __CancelBtnHandler(self, button: 'Button', action: str): 

115 if action == 'Pressed': 

116 button.SetState(1) 

117 elif action == 'Released': 

118 self.HidePINMenu() 

119 button.SetState(0) 

120 

121 def __StartBtnHandler(self, button: 'Button', action: str): 

122 button.SetState(0) 

123 self.ShowPINMenu()

127 def __MaskPIN(self) -> None: 

128 """Generates and sets Masked PIN feedback""" 

129 

130 mask = "" 

131 while (len(mask) < len(self.__CurrentPIN)): 

132 mask = mask + "*" 

133 self.__PINLbl.SetText(mask)
