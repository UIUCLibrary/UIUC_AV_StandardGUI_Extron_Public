507 def __GetCmdQualifier(self, command: str, block: str, channel: Union[str, int]) -> dict: 

508 if self.DSP.Manufacturer == "Biamp": 

509 return {'Instance Tag': str(block), 'Channel': str(channel)} 

510 elif self.DSP.Manufacturer == "BSS": 

511 if command == 'Gain': 

512 id = 4 + ((int(channel) - 1) * 6) 

513 elif command == 'PhantomSwitch': 

514 id = 5 + ((int(channel) - 1) * 6) 

515 else: 

516 raise ValueError('BSS DSP Command ({}) not configured'.format(command)) 

517 return {'HiQAddress': str(block), 'ID': id} 

518 else: 

519 raise ValueError('DSP Manufacturer ({}) not configured'.format(self.DSP.Manufacturer))

616 def AdjustLevel(self, level, direction: str='up', step: str='small'): 

617 if type(level) is not Level: 

618 raise TypeError('Level must be a Level object') 

619 if type(direction) is not str: 

620 raise TypeError('Direction must be a string') 

621 if type(step) is not str: 

622 raise TypeError('Step must be a string') 

623 

624 if step == 'small': 

625 s = self.__StepSm 

626 elif step == 'large': 

627 s = self.__StepLg 

628 else: 

629 raise ValueError('Step must be either "large" or "small"') 

630 

631 if direction == 'up': 

632 newLevel = level.Level + s 

633 elif direction == 'down': 

634 newLevel = level.Level - s 

635 else: 

636 raise ValueError('Direction must either be "up" or "down"') 

637 

638 level.SetLevel(newLevel) 

639 

640 return newLevel

642 def AdjustGain(self, gain, direction: str='up'): 

643 if type(gain) is not Label: 

644 raise TypeError('Level must be a Gain Label object') 

645 if type(direction) is not str: 

646 raise TypeError('Direction must be a string') 

647 

648 mfg = self.DSP.Manufacturer 

649 inputIndex = self.__GainControlLists['dBLbl'].index(gain) 

650 

651 if direction == 'up': 

652 newGain = gain.GainVal + 6 

653 elif direction == 'down': 

654 newGain = gain.GainVal - 6 

655 else: 

656 raise ValueError('Direction must either be "up" or "down"') 

657 

658 downEnable = True 

659 upEnable = True 

660 

661 if newGain <= 0: 

662 # disable down arrow 

663 downEnable = False 

664 newGain = 0 

665 

666 if mfg == 'Biamp': 

667 newLevel = newGain 

668 if newGain >= 66: 

669 upEnable = False 

670 elif mfg == 'BSS': 

671 # BSS require gain in a percentage value 

672 newLevel = (newGain / 48) * 100 

673 if newGain >= 48: 

674 upEnable = False 

675 

676 

677 gain.GainVal = newGain 

678 

679 self.__GainControlLists['down'][inputIndex].SetEnable(downEnable) 

680 if downEnable: 

681 self.__GainControlLists['down'][inputIndex].SetState(0) 

682 else: 

683 self.__GainControlLists['down'][inputIndex].SetState(2) 

684 

685 self.__GainControlLists['up'][inputIndex].SetEnable(upEnable) 

686 if upEnable: 

687 self.__GainControlLists['up'][inputIndex].SetState(0) 

688 else: 

689 self.__GainControlLists['up'][inputIndex].SetState(2) 

690 

691 if newGain == 0: 

692 gain.SetText("0dB") 

693 elif newGain > 0: 

694 gain.SetText('+{}dB'.format(newGain)) 

695 

696 return newLevel
