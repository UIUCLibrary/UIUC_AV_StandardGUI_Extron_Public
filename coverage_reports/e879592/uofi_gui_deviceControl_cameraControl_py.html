257 def SavePresetStates(self): 

258 # only need to save the preset names, presets are stored presistently on camera 

259 if File.Exists(self.__PresetsFilePath): 

260 # file exists -> read file to object, modify object, save object to file 

261 #### read file to object 

262 presetsFile = File(self.__PresetsFilePath, 'rt') 

263 presetString = presetsFile.read() 

264 presetObj = json.loads(presetString) 

265 presetsFile.close() 

266 

267 #### modify object 

268 for cam in self.Cameras.values(): 

269 if cam['Id'] not in presetObj: 

270 presetObj[cam['Id']] = {} 

271 

272 for i in range(1, 4): # Preset 0 is home, Presets 1-3 are displayed buttons 

273 if i in cam['Hw'].Presets: 

274 presetObj[cam['Id']][i] = cam['Hw'].Presets[i] 

275 else: 

276 presetObj[cam['Id']][i] = None 

277 

278 #### save object to file 

279 presetsFile = File(self.__PresetsFilePath, 'wt') 

280 presetsFile.write(json.dumps(presetObj)) 

281 presetsFile.close() 

282 else: 

283 # file does not exist -> create object, save object to file 

284 #### create object 

285 presetObj = {} 

286 

287 for cam in self.Cameras.values(): 

288 # Log('Cam Info: {}'.format(cam)) 

289 presetObj[cam['Id']] = {} 

290 

291 for i in range(1, 4): # Preset 0 is home, Presets 1-3 are displayed buttons 

292 if i in cam['Hw'].Presets: 

293 presetObj[cam['Id']][i] = cam['Hw'].Presets[i] 

294 else: 

295 presetObj[cam['Id']][i] = None 

296 

297 #### save object to file 

298 presetsFile = File(self.__PresetsFilePath, 'xt') 

299 presetsFile.write(json.dumps(presetObj)) 

300 presetsFile.close()

302 def LoadPresetStates(self): 

303 # only need to load the preset names, presets are stored presistently on camera 

304 if File.Exists(self.__PresetsFilePath): 

305 #### read file to object 

306 presetsFile = File(self.__PresetsFilePath, 'rt') 

307 presetString = presetsFile.read() 

308 presetObj = json.loads(presetString) 

309 # Log('JSON Obj: {}'.format(presetObj)) 

310 presetsFile.close() 

311 

312 #### iterate over objects and load presets 

313 for cam in self.Cameras.values(): 

314 if cam['Id'] in presetObj: 

315 for i in presetObj[cam['Id']]: 

316 # watch the typing here, rest of module expects i to be an int but i is a string in presetObj 

317 if presetObj[cam['Id']][str(i)] is not None: 

318 cam['Hw'].Presets[int(i)] = presetObj[cam['Id']][str(i)] 

319 

320 else: 

321 Log('No presets file exists')
